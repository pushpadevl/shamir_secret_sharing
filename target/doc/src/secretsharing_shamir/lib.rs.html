86    pub fn new( prime_size: BitSize, use_fixed_prime:bool,threshold: u8, secret: &BigUint) -> Result<Self, Error> {
87        if threshold <= 1 {
88            return Err(Error::ThresholdTooSmall);
89        }
90            let prim = if use_fixed_prime {
91                                    prime_size.fixed_prime()
92                                } else { 
93                                    prime_size.new_prime()
94                                };
95            
96            let mut instance = SS {
97                prime: prim,
98                polynomial:Vec::new(),
99            };
100            
101            let secret_mod = secret % &instance.prime;
102            instance.gen_polynomial(secret_mod,threshold, prime_size);
103            
104            Ok(instance)
105    }

111    pub fn gen_shares(&mut self, points:&Vec<BigUint>) -> Vec<Share> {
112        let n:usize = points.len();
113        let mut shares:Vec<Share> = Vec::with_capacity(n);
114
115        for i in 0..(n) {
116            shares.push(
117                Share{
118                    X:points[i].clone(),
119                    Y:self.eval_px_at_xi(&points[i])
120                }
121            );
122        }
123        shares

126    fn eval_px_at_xi(&self, x:&BigUint)-> BigUint{ 
127        let mut y:BigUint = (self.polynomial)[0].clone(); //init 
128        let mut x_pow:BigUint = x.clone(); 
129        for i in 1..(self.polynomial).len() {
130            let tmp = (&x_pow * (self.polynomial)[i].clone() ) % &self.prime;
131            y = (&y + &tmp) % &self.prime; 
132            x_pow = ( &x_pow * x) % &self.prime; 
133        } 
134        y 
135    }

137    pub fn gen_polynomial(&mut self,secret:BigUint, degree:u8, prime_size: BitSize){
138
139        let p = &self.prime;
140        self.polynomial.push(secret % &self.prime); // secret consumed here
141
142        for _ in 0..(degree -1) {
143            let coeff:BigUint = prime_size.n_bit_random() % p; 
144            self.polynomial.push(coeff);
145        } 
146        
147        let mut leading_coeff;
148        loop {
149            leading_coeff = prime_size.n_bit_random() % p;
150            if !leading_coeff.is_zero() {
151                break;
152            } 
153        }
154        self.polynomial.push(leading_coeff);
155    }

157    fn gcd(x:&BigUint,y:&BigUint) -> Result<BigUint,Error>{
158         if x.is_zero() || y.is_zero() {
159        return Err(Error::ZeroInputGCD);
160        }
161
162        let mut a = x.clone();
163        let mut b = y.clone();
164
165        while !b.is_zero() {
166            let tmp = b.clone();
167            b = a % &b;
168            a = tmp;
169        }
170
171        Ok(a)
172    }

174    fn inv_modp(prime:&BigUint,a: &BigUint) -> Result<BigUint,Error>{ 
175        let aa = (prime + a) % prime;
176        if !SS::gcd(&aa,prime).unwrap().is_one() {
177            return Err(Error::NotCoprimes);
178        }   
179        match aa.modinv(prime) {
180            Some(inv) => Ok(inv),
181            None => Err(Error::NotCoprimes)
182        }

187    pub fn reconstruct_secret(prime:&BigUint, shares:&Vec<Share>) -> BigUint{ 
188        /* Comment: Does not have self,user should be able to try arbitrary share values to check if it matches.
189           2. Also, it should work without an instance of SS, as that is majorly used for generating_shares.
190        */ 
191        let n = shares.len();
192        let mut res:BigUint = BigUint::zero();
193 
194        for i in 0..n {
195            let xi = &shares[i].X;
196            let yi = &shares[i].Y;
197
198            let mut num=BigUint::one();
199            let mut den=BigUint::one();
200            
201            for j in 0..n {
202                if i == j { continue; }
203                let xj = &shares[j].X;
204
205                // numerator term: (0 - xj) mod p == (p - xj) mod p
206                let term_num = (prime - xj) % prime;
207                num = (num * term_num) % prime;
208
209                // denominator term: (xi - xj) mod p == (xi + (p - xj)) mod p
210                let term_den = (xi + ((prime - xj) % prime)) % prime;
211                // x's should not be equal, other inverse operation would panic
212                den = (den * term_den) % prime;
213            }
214
215            // inv_modp should return the modular inverse of `den` modulo `p`
216            let den_inv = SS::inv_modp(prime, &den).unwrap();
217            let li0 = (num * den_inv) % prime; // lagrange interpolation
218
219            // accumulate: y_i * lambda_i(0)
220            res = (res + (yi * &li0) % prime) % prime;
221        }
222        
223        res
224    }
